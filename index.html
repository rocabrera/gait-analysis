def check3DNormOrt(e1, e2, e3, erro = 1e-6):
    print('----------------------------------------------')
    
    print('Checando normalidade dos versores:')
    print('e1: {}'.format(np.all( np.abs(np.linalg.norm(e1, axis = 1) -1) < erro)))    
    print('e2: {}'.format(np.all( np.abs(np.linalg.norm(e1, axis = 1) -1) < erro)))        
    print('e3: {}\n'.format(np.all( np.abs(np.linalg.norm(e1, axis = 1) -1) < erro)))  
    print('Checando ortogonalidade dos versores:')
    print('e1 X e2: {}'.format(np.all(np.sum(e1*e2, axis = 1) < erro )))
    print('e1 X e3: {}'.format(np.all(np.sum(e1*e2, axis = 1) < erro )))
    print('e2 X e3: {}'.format(np.all(np.sum(e1*e2, axis = 1) < erro )))
    
    print('----------------------------------------------')
    
def bodyBase3D(p1, p2, p3):
    OM = (p1+p2)/2
    V1_3D = OM - p3
    e1_3D = V1_3D/np.linalg.norm(V1_3D, axis = 1, keepdims=True)
    #
    V2_3D = p2 - OM
    e2_3D = np.cross(V2_3D, e1_3D)
    e2_3D = e2_3D/np.linalg.norm(e2_3D,axis=1,keepdims = True)
    
    e3_3D = np.cross(e1_3D, e2_3D)
    e3_3D = e3_3D/np.linalg.norm(e3_3D,axis=1,keepdims = True)

    check3DNormOrt(e1_3D , e2_3D, e3_3D)
    
    return e1_3D, e2_3D, e3_3D
    

def plotImage(omega, alpha):
    
    data = [omega, alpha]
    
    t_omega = np.arange(len(omega))*dt
    t_alpha = np.arange(len(alpha))*dt
    t = [t_omega, t_alpha]
    
    titles  = ['Velocidade Angular (Local)', 'Aceleração Angular (Local)']
    ylabels = [r'Velocidade $ \frac{rad}{s}$', r'Aceleração $ \frac{rad}{s^2}$']
    
    fig, axs = plt.subplots(1,2, figsize = (20,8))
    
    for ax, title, ylabel, x, y in zip(axs.flatten(), titles, ylabels, t, data):
        ax.plot(x, y)
        ax.set_title(title, fontsize = 18)
        ax.set_ylabel(ylabel)
        ax.set_xlabel('Tempo (Segundos)')
        ax.legend(('e1', 'e2', 'e3'))

    plt.show()

Fazendo as bases locais¶

A base é feita a partir de 3 pontos.

  • O primeiro versor foi definido na direção do segmento.

  • Para o cálculo do segundo versor utiliza-se um vetor auxiliar que é perpendicula a direção do movimento.

  • O segundo versor foi definido como o produto vetorial entre o primeiro versor e o vetor auxiliar.

  • O terceiro versor é definido como o produto vetorial entre o primeiro e segundo versor.

innerProduct = lambda x,y: np.sum(x*y, axis=1, keepdims = True)

A velocidade angular do movimento 3D pode ser definida através da seguinte equação:

\begin{equation} \vec{\boldsymbol{\omega}} = \left(\frac{d\hat{\boldsymbol{e_2}}}{dt}\cdot \hat{\boldsymbol{e_3}}\right) \hat{\boldsymbol{e_1}} + \left(\frac{d\hat{\boldsymbol{e_3}}}{dt}\cdot \hat{\boldsymbol{e_1}}\right) \hat{\boldsymbol{e_2}} + \left(\frac{d\hat{\boldsymbol{e_1}}}{dt}\cdot \hat{\boldsymbol{e_2}}\right) \hat{\boldsymbol{e_3}} \label{eq:angvel} \end{equation}

Dessa forma é necessário calcular a derivada dos versores da base local
